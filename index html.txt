    <script>
        // Mathematical Constants based on MPOB industrial processing averages at 90°C
        const L = 1000;   // Max attainable clarified oil capacity (Liters)
        const k = 0.85;   // Rate coefficient matching high-temperature separation
        const t0 = 4.0;   // Separation midpoint inflection point (Hours)

        // Generate Data Points (0 to 10 hours, steps of 0.5)
        const timeLabels = [];
        const volumeData = [];
        const derivativeData = [];

        for (let t = 0; t <= 10; t += 0.5) {
            timeLabels.push(t);
            // Logistic function V(t)
            const V = L / (1 + Math.exp(-k * (t - t0)));
            volumeData.push(V);
            // First Derivative V'(t) = k * V * (1 - V/L)
            const V_prime = k * V * (1 - (V / L));
            derivativeData.push(V_prime);
        }
